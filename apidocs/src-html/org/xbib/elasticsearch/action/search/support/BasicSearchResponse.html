002package org.xbib.elasticsearch.action.search.support;
003
004import org.elasticsearch.action.search.SearchResponse;
005import org.elasticsearch.common.xcontent.ToXContent;
006import org.elasticsearch.common.xcontent.XContentBuilder;
007import org.elasticsearch.common.xcontent.json.JsonXContent;
008
009import java.io.IOException;
010import java.io.OutputStream;
011
012/**
013 * Helper class for Elasticsearch responses
014 */
015public class BasicSearchResponse {
016
017    private SearchResponse searchResponse;
018
019    public BasicSearchResponse setResponse(SearchResponse response) {
020        this.searchResponse = response;
021        return this;
022    }
023
024    public SearchResponse getResponse() {
025        return searchResponse;
026    }
027
028    public long tookInMillis() {
029        return searchResponse.getTookInMillis();
030    }
031
032    public long totalHits() {
033        return searchResponse.getHits().getTotalHits();
034    }
035
036    public BasicSearchResponse toJson(OutputStream out) throws IOException {
037        if (out == null) {
038            return this;
039        }
040        if (searchResponse == null) {
041            out.write(jsonErrorMessage("no response"));
042            return this;
043        }
044        XContentBuilder jsonBuilder = new XContentBuilder(JsonXContent.jsonXContent, out);
045        jsonBuilder.startObject();
046        searchResponse.toXContent(jsonBuilder, ToXContent.EMPTY_PARAMS);
047        jsonBuilder.endObject();
048        jsonBuilder.close();
049        return this;
050    }
051
052    private static byte[] jsonErrorMessage(String message) {
053        return ("{\"error\":500,\"message\":\"" + message + "\"}").getBytes();
054    }
055
056
057}
