001package org.xbib.elasticsearch.action.support.replication.replica;
002
003import org.elasticsearch.action.ActionRequest;
004import org.elasticsearch.action.ActionRequestValidationException;
005import org.elasticsearch.common.io.stream.StreamInput;
006import org.elasticsearch.common.io.stream.StreamOutput;
007import org.elasticsearch.common.unit.TimeValue;
008
009import java.io.IOException;
010import java.util.concurrent.TimeUnit;
011
012import static org.elasticsearch.action.ValidateActions.addValidationError;
013
014public abstract class ReplicaShardOperationRequest<T extends ReplicaShardOperationRequest> extends ActionRequest<T> {
015
016    public static final TimeValue DEFAULT_TIMEOUT = new TimeValue(1, TimeUnit.MINUTES);
017
018    protected TimeValue timeout = DEFAULT_TIMEOUT;
019
020    protected String index;
021
022    private boolean threadedOperation = true;
023
024    protected ReplicaShardOperationRequest() {
025    }
026
027    public ReplicaShardOperationRequest(ActionRequest request) {
028        super(request);
029    }
030
031    public ReplicaShardOperationRequest(T request) {
032        super(request);
033        this.timeout = request.timeout();
034        this.index = request.index();
035        this.threadedOperation = request.operationThreaded();
036    }
037
038    public final boolean operationThreaded() {
039        return threadedOperation;
040    }
041
042    @SuppressWarnings("unchecked")
043    public final T operationThreaded(boolean threadedOperation) {
044        this.threadedOperation = threadedOperation;
045        return (T) this;
046    }
047
048    @SuppressWarnings("unchecked")
049    public final T timeout(TimeValue timeout) {
050        this.timeout = timeout;
051        return (T) this;
052    }
053
054    public final T timeout(String timeout) {
055        return timeout(TimeValue.parseTimeValue(timeout, null));
056    }
057
058    public TimeValue timeout() {
059        return timeout;
060    }
061
062    public String index() {
063        return index;
064    }
065
066    @SuppressWarnings("unchecked")
067    public final T index(String index) {
068        this.index = index;
069        return (T) this;
070    }
071
072    @Override
073    public ActionRequestValidationException validate() {
074        ActionRequestValidationException validationException = null;
075        if (index == null) {
076            validationException = addValidationError("index is missing", null);
077        }
078        return validationException;
079    }
080
081    @Override
082    public void readFrom(StreamInput in) throws IOException {
083        super.readFrom(in);
084        timeout = TimeValue.readTimeValue(in);
085        index = in.readString();
086    }
087
088    @Override
089    public void writeTo(StreamOutput out) throws IOException {
090        super.writeTo(out);
091        timeout.writeTo(out);
092        out.writeString(index);
093    }
094
095}
