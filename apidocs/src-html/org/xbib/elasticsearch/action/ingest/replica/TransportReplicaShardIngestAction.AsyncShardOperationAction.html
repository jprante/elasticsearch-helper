126    protected IngestReplicaShardResponse shardOperationOnReplica(ReplicaOperationRequest shardRequest) {
127        final long t0 = shardRequest.startTime();
128        final IngestReplicaShardRequest request = shardRequest.request();
129        final IndexShard indexShard = indicesService.indexServiceSafe(shardRequest.request().index()).shardSafe(shardRequest.shardId());
130        int successCount = 0;
131        List<IngestActionFailure> failure = new LinkedList<>();
132        int size = request.actionRequests().size();
133        for (int i = 0; i < size; i++) {
134            ActionRequest actionRequest = request.actionRequests().get(i);
135            if (actionRequest == null) {
136                continue;
137            }
138            if (actionRequest instanceof IndexRequest) {
139                IndexRequest indexRequest = (IndexRequest) actionRequest;
140                try {
141                    indexOperationOnReplica(indexShard, indexRequest);
142                    successCount++;
143                } catch (Throwable e) {
144                    failure.add(new IngestActionFailure(request.ingestId(), request.shardId(), ExceptionsHelper.detailedMessage(e)));
145                }
146            } else if (actionRequest instanceof DeleteRequest) {
147                DeleteRequest deleteRequest = (DeleteRequest) actionRequest;
148                try {
149                    Engine.Delete delete = indexShard.prepareDelete(deleteRequest.type(),
150                            deleteRequest.id(),
151                            deleteRequest.version(),
152                            deleteRequest.versionType(),
153                            Engine.Operation.Origin.REPLICA);
154                    indexShard.delete(delete);
155                    successCount++;
156                } catch (Throwable e) {
157                    failure.add(new IngestActionFailure(request.ingestId(), request.shardId(), ExceptionsHelper.detailedMessage(e)));
158                }
159            }
160        }
161        return new IngestReplicaShardResponse(request.ingestId(), request.shardId(), shardRequest.replicaId(),
162                successCount, System.currentTimeMillis() - t0, failure);
163    }
164
165    private void indexOperationOnReplica(IndexShard indexShard, IndexRequest indexRequest) {
166        SourceToParse sourceToParse = SourceToParse.source(SourceToParse.Origin.REPLICA, indexRequest.source())
167                .type(indexRequest.type())
168                .id(indexRequest.id())
169                .routing(indexRequest.routing())
170                .parent(indexRequest.parent())
171                .timestamp(indexRequest.timestamp())
172                .ttl(indexRequest.ttl());
173        if (indexRequest.opType() == IndexRequest.OpType.INDEX) {
174            Engine.Index index = indexShard.prepareIndex(sourceToParse,
175                    indexRequest.version(),
176                    indexRequest.versionType(),
177                    Engine.Operation.Origin.REPLICA,
178                    false);
179            indexShard.index(index);
180        } else {
181            Engine.Create create = indexShard.prepareCreate(sourceToParse,
182                    indexRequest.version(),
183                    indexRequest.versionType(),
184                    Engine.Operation.Origin.REPLICA,
185                    false,
186                    indexRequest.autoGeneratedId());
187            indexShard.create(create);
188        }
189    }

198        return true;

201    boolean ignoreReplicaException(Throwable e) {
202        if (TransportActions.isShardNotAvailableException(e)) {
203            return true;
204        }
205        Throwable cause = ExceptionsHelper.unwrapCause(e);
206        return cause instanceof VersionConflictEngineException;
207    }

209    protected class AsyncShardOperationAction {
210
211        private final IngestReplicaShardRequest request;
212
213        private final ActionListener<ReplicaOperationResponse> listener;
214
215        private final ReplicaOperationResponse response;
216
217        private volatile ShardIterator shardIt;
218
219        private volatile ClusterStateObserver observer;
220
221        AsyncShardOperationAction(IngestReplicaShardRequest request, ActionListener<ReplicaOperationResponse> listener) {
222            this.request = request;
223            this.listener = listener;
224            this.response = newReplicaResponseInstance();
225        }
226
227        public void start() {
228            observer = new ClusterStateObserver(clusterService, request.timeout(), logger);
229            doStart();
230        }
231
232        protected boolean doStart() throws ElasticsearchException {
233            try {
234                ClusterBlockException blockException = checkGlobalBlock(observer.observedState(), request);
235                if (blockException != null) {
236                    if (blockException.retryable()) {
237                        if (logger.isTraceEnabled()) {
238                            logger.trace("cluster is blocked ({}), scheduling a retry", blockException.getMessage());
239                        }
240                        retry(blockException);
241                        return false;
242                    } else {
243                        throw blockException;
244                    }
245                }
246                if (!resolveRequest(observer.observedState(), request)) {
247                    return true;
248                }
249                blockException = checkRequestBlock(observer.observedState(), request);
250                if (blockException != null) {
251                    if (blockException.retryable()) {
252                        if (logger.isTraceEnabled()) {
253                            logger.trace("cluster is blocked ({}), scheduling a retry", blockException.getMessage());
254                        }
255                        retry(blockException);
256                        return false;
257                    } else {
258                        throw blockException;
259                    }
260                }
261                shardIt = shards(observer.observedState(), request);
262            } catch (Throwable e) {
263                logger.error(e.getMessage(), e);
264                listener.onFailure(e);
265                return true;
266            }
267            if (shardIt.size() == 0) {
268                if (logger.isTraceEnabled()) {
269                    logger.trace("no shard instances known for shard [{}], scheduling a retry", shardIt.shardId());
270                }
271                retry(null);
272                return false;
273            }
274            ShardRouting shard;
275            ClusterState newState = clusterService.state();
276            ShardRouting newLeaderShard = null;
277            if (observer.observedState() != newState) {
278                shardIt.reset();
279                ShardRouting originalLeaderShard = null;
280                while ((shard = shardIt.nextOrNull()) != null) {
281                    if (shard.primary()) {
282                        originalLeaderShard = shard;
283                        break;
284                    }
285                }
286                if (originalLeaderShard == null || !originalLeaderShard.active()) {
287                    throw new ElasticsearchException("unexpected state, failed to find leader shard on an index operation that succeeded");
288                }
289                observer.reset(newState);
290                shardIt = shards(newState, request);
291                while ((shard = shardIt.nextOrNull()) != null) {
292                    if (shard.primary()) {
293                        if (originalLeaderShard.currentNodeId().equals(shard.currentNodeId())) {
294                            newLeaderShard = null;
295                        } else {
296                            newLeaderShard = shard;
297                        }
298                        break;
299                    }
300                }
301                shardIt.reset();
302            }
303            int replicaCounter = shardIt.assignedReplicasIncludingRelocating();
304            if (newLeaderShard != null) {
305                replicaCounter++;
306            }
307            if (replicaCounter == 0) {
308                return true;
309            }
310            final IndexMetaData indexMetaData = observer.observedState().metaData().index(request.index());
311            if (newLeaderShard != null) {
312                logger.debug("we have a new leader shard?");
313                //performOnReplica(newLeaderShard, replicaCounter, newLeaderShard.currentNodeId(), indexMetaData);
314            }
315            List<ReplicaInfo> infos = new LinkedList<>();
316            int replicaLevel = 1;
317            shardIt.reset();
318            while ((shard = shardIt.nextOrNull()) != null) {
319                if (shard.unassigned()) {
320                    continue;
321                }
322                boolean doOnlyOnRelocating = false;
323                if (shard.primary()) {
324                    if (shard.relocating()) {
325                        doOnlyOnRelocating = true;
326                    } else {
327                        continue;
328                    }
329                }
330                String nodeId = !doOnlyOnRelocating ? shard.currentNodeId() : shard.relocating() ? shard.relocatingNodeId() : null;
331                if (nodeId == null) {
332                    continue;
333                }
334                infos.add(new ReplicaInfo(nodeId, shard, replicaLevel));
335                replicaLevel++;
336            }
337            final AtomicLong replicas = new AtomicLong(infos.size());
338            for (ReplicaInfo info : infos) {
339                final ReplicaOperationRequest replicaRequest = new ReplicaOperationRequest(System.currentTimeMillis(),
340                        request.index(), info.shardRouting().shardId().id(), info.replicaLevel(), request);
341                final String nodeId = info.nodeId();
342                final ShardRouting shardRouting = info.shardRouting();
343                if (!observer.observedState().nodes().nodeExists(nodeId)) {
344                    if (logger.isDebugEnabled()) {
345                        logger.debug("node {} no longer exists", nodeId);
346                    }
347                } else {
348                    if (nodeId.equals(observer.observedState().nodes().localNodeId())) {
349                        if (request.operationThreaded()) {
350                            try {
351                                threadPool.executor(executor).execute(new AbstractRunnable() {
352
353                                    @Override
354                                    public void onFailure(Throwable t) {
355                                        logger.error(t.getMessage(), t);
356                                        failReplicaIfNeeded(shardRouting.index(), shardRouting.id(), t);
357                                    }
358
359                                    @Override
360                                    protected void doRun() throws Exception {
361                                        response.add(shardOperationOnReplica(replicaRequest));
362                                        if (replicas.decrementAndGet() == 0) {
363                                            listener.onResponse(response);
364                                        }
365                                    }
366
367                                    @Override
368                                    public boolean isForceExecution() {
369                                        return true;
370                                    }
371                                });
372                            } catch (Throwable e) {
373                                logger.error(e.getMessage(), e);
374                                failReplicaIfNeeded(shard.index(), shard.id(), e);
375                            }
376                        } else {
377                            try {
378                                response.add(shardOperationOnReplica(replicaRequest));
379                                if (replicas.decrementAndGet() == 0) {
380                                    listener.onResponse(response);
381                                }
382                            } catch (Throwable e) {
383                                logger.error(e.getMessage(), e);
384                                failReplicaIfNeeded(shard.index(), shard.id(), e);
385                            }
386                        }
387                    } else {
388                        final DiscoveryNode node = observer.observedState().nodes().get(nodeId);
389                        transportService.sendRequest(node, transportAction, replicaRequest, transportOptions, new BaseTransportResponseHandler<IngestReplicaShardResponse>() {
390                            @Override
391                            public IngestReplicaShardResponse newInstance() {
392                                return newResponseInstance();
393                            }
394
395                            @Override
396                            public String executor() {
397                                return ThreadPool.Names.SAME;
398                            }
399
400                            @Override
401                            public void handleResponse(IngestReplicaShardResponse result) {
402                                response.add(result);
403                                if (replicas.decrementAndGet() == 0) {
404                                    listener.onResponse(response);
405                                }
406                            }
407
408                            @Override
409                            public void handleException(TransportException exp) {
410                                logger.error(exp.getMessage(), exp);
411                                if (!ignoreReplicaException(exp.unwrapCause())) {
412                                    logger.warn("failed to perform " + transportAction + " on replica " + node + shardIt.shardId(), exp);
413                                    shardStateAction.shardFailed(shardRouting, indexMetaData.getUUID(),
414                                            "failed to perform [" + transportAction + "] on replica, message [" + ExceptionsHelper.detailedMessage(exp) + "]");
415                                } else {
416                                    logger.error(exp.getMessage(), exp);
417                                }
418                                listener.onFailure(exp);
419                            }
420                        });
421                    }
422                }
423            }
424            return true;
425        }
426
427        private void retry(@Nullable final Throwable failure) {
428            if (observer.isTimedOut()) {
429                listener.onFailure(failure);
430                return;
431            }
432            request.operationThreaded(true);
433            observer.waitForNextChange(new ClusterStateObserver.Listener() {
434                @Override
435                public void onNewClusterState(ClusterState state) {
436                    doStart();
437                }
438
439                @Override
440                public void onClusterServiceClose() {
441                    listener.onFailure(new NodeClosedException(clusterService.localNode()));
442                }
443
444                @Override
445                public void onTimeout(TimeValue timeout) {
446                    if (doStart()) {
447                        return;
448                    }
449                    raiseTimeoutFailure(timeout, failure);
450                }
451            });
452        }
453
454        void raiseTimeoutFailure(TimeValue timeout, @Nullable Throwable failure) {
455            if (failure == null) {
456                if (shardIt == null) {
457                    failure = new UnavailableShardsException(null, "no available shards, timeout waiting for [" + timeout + "], request: " + request.toString());
458                } else {
459                    failure = new UnavailableShardsException(shardIt.shardId(), "[" + shardIt.size() + "] shardIt, [" + shardIt.sizeActive() + "] active : Timeout waiting for [" + timeout + "], request: " + request.toString());
460                }
461            }
462            listener.onFailure(failure);
463        }
464    }

466    private void failReplicaIfNeeded(String index, int shardId, Throwable t) {
467        if (!ignoreReplicaException(t)) {
468            IndexService indexService = indicesService.indexService(index);
469            if (indexService == null) {
470                if (logger.isDebugEnabled()) {
471                    logger.debug("ignoring failed replica [{}][{}] because index was already removed", index, shardId);
472                }
473                return;
474            }
475            IndexShard indexShard = indexService.shard(shardId);
476            if (indexShard == null) {
477                if (logger.isDebugEnabled()) {
478                    logger.debug("ignoring failed replica [{}][{}] because index was already removed", index, shardId);
479                }
480                return;
481            }
482            indexShard.failShard(transportAction + " failed on replica", t);
483        }
484    }

513    public class ReplicaOperationResponse extends ActionResponse {
514
515        private final List<IngestReplicaShardResponse> responses = new LinkedList<>();
516
517        public ReplicaOperationResponse add(IngestReplicaShardResponse response) {
518            responses.add(response);
519            return this;
520        }
521
522        public List<IngestReplicaShardResponse> responses() {
523            return responses;
524        }
525
526        @Override
527        public void readFrom(StreamInput in) throws IOException {
528            super.readFrom(in);
529            for (int i = 0; i < in.readVInt(); i++) {
530                IngestReplicaShardResponse response = newResponseInstance();
531                response.readFrom(in);
532                responses.add(response);
533            }
534        }
535
536        @Override
537        public void writeTo(StreamOutput out) throws IOException {
538            super.writeTo(out);
539            out.writeVInt(responses.size());
540            for (IngestReplicaShardResponse response : responses) {
541                response.writeTo(out);
542            }
543        }
544
545        public String toString() {
546            return responses.toString();
547        }
548    }

550    class ReplicaOperationTransportHandler implements TransportRequestHandler<ReplicaOperationRequest> {
551
552        @Override
553        public void messageReceived(final ReplicaOperationRequest request, final TransportChannel channel) throws Exception {
554            try {
555                ActionResponse response = shardOperationOnReplica(request);
556                channel.sendResponse(response);
557            } catch (Throwable t) {
558                logger.error(t.getMessage(), t);
559                channel.sendResponse(t);
560            }
561        }
