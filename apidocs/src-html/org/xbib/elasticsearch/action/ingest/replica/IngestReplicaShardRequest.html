001package org.xbib.elasticsearch.action.ingest.replica;
002
003import org.elasticsearch.ElasticsearchException;
004import org.elasticsearch.action.ActionRequest;
005import org.elasticsearch.action.ActionRequestValidationException;
006import org.elasticsearch.action.delete.DeleteRequest;
007import org.elasticsearch.action.index.IndexRequest;
008import org.elasticsearch.action.support.IndicesOptions;
009import org.elasticsearch.common.io.stream.StreamInput;
010import org.elasticsearch.common.io.stream.StreamOutput;
011import org.elasticsearch.common.unit.TimeValue;
012import org.elasticsearch.index.shard.ShardId;
013
014import java.io.IOException;
015import java.util.LinkedList;
016import java.util.List;
017import java.util.concurrent.TimeUnit;
018
019import static org.elasticsearch.action.ValidateActions.addValidationError;
020
021public class IngestReplicaShardRequest extends ActionRequest<IngestReplicaShardRequest> {
022
023    public static final TimeValue DEFAULT_TIMEOUT = new TimeValue(1, TimeUnit.MINUTES);
024
025    private TimeValue timeout = DEFAULT_TIMEOUT;
026
027    private String index;
028
029    private boolean threadedOperation = true;
030
031    private long ingestId;
032
033    private ShardId shardId;
034
035    private List<ActionRequest> actionRequests = new LinkedList<>();
036
037    public IngestReplicaShardRequest() {
038    }
039
040    public IngestReplicaShardRequest(long ingestId, ShardId shardId, List<ActionRequest> actionRequests) {
041        this.index = shardId.index().name();
042        this.ingestId = ingestId;
043        this.shardId = shardId;
044        this.actionRequests = actionRequests;
045    }
046
047    public long ingestId() {
048        return ingestId;
049    }
050
051    public ShardId shardId() {
052        return shardId;
053    }
054
055    public List<ActionRequest> actionRequests() {
056        return actionRequests;
057    }
058
059    public final boolean operationThreaded() {
060        return threadedOperation;
061    }
062
063    @SuppressWarnings("unchecked")
064    public final IngestReplicaShardRequest operationThreaded(boolean threadedOperation) {
065        this.threadedOperation = threadedOperation;
066        return this;
067    }
068
069    @SuppressWarnings("unchecked")
070    public final IngestReplicaShardRequest timeout(TimeValue timeout) {
071        this.timeout = timeout;
072        return this;
073    }
074
075    public TimeValue timeout() {
076        return timeout;
077    }
078
079    public String index() {
080        return index;
081    }
082
083    @SuppressWarnings("unchecked")
084    public final IngestReplicaShardRequest index(String index) {
085        this.index = index;
086        return this;
087    }
088
089    public String[] indices() {
090        return new String[]{index};
091    }
092
093    public IndicesOptions indicesOptions() {
094        return IndicesOptions.strictSingleIndexNoExpandForbidClosed();
095    }
096
097    @Override
098    public ActionRequestValidationException validate() {
099        ActionRequestValidationException validationException = null;
100        if (index == null) {
101            validationException = addValidationError("index is missing", null);
102        }
103        return validationException;
104    }
105
106    @Override
107    public void writeTo(StreamOutput out) throws IOException {
108        super.writeTo(out);
109        out.writeString(index);
110        timeout.writeTo(out);
111        out.writeLong(ingestId);
112        shardId.writeTo(out);
113        out.writeVInt(actionRequests.size());
114        for (ActionRequest actionRequest : actionRequests) {
115            if (actionRequest == null) {
116                out.writeBoolean(false);
117                continue;
118            }
119            out.writeBoolean(true);
120            if (actionRequest instanceof IndexRequest) {
121                out.writeBoolean(true);
122            } else if (actionRequest instanceof DeleteRequest) {
123                out.writeBoolean(false);
124            } else {
125                throw new ElasticsearchException("action request not supported: " + actionRequest.getClass().getName());
126            }
127            actionRequest.writeTo(out);
128        }
129    }
130
131    @Override
132    public void readFrom(StreamInput in) throws IOException {
133        super.readFrom(in);
134        index = in.readString();
135        timeout = TimeValue.readTimeValue(in);
136        ingestId = in.readLong();
137        shardId = ShardId.readShardId(in);
138        int size = in.readVInt();
139        actionRequests = new LinkedList<>();
140        for (int i = 0; i < size; i++) {
141            boolean exists = in.readBoolean();
142            if (exists) {
143                boolean b = in.readBoolean();
144                if (b) {
145                    IndexRequest indexRequest = new IndexRequest();
146                    indexRequest.readFrom(in);
147                    actionRequests.add(indexRequest);
148                } else {
149                    DeleteRequest deleteRequest = new DeleteRequest();
150                    deleteRequest.readFrom(in);
151                    actionRequests.add(deleteRequest);
152                }
153            } else {
154                actionRequests.add(null);
155            }
156        }
157    }
158}
