002    package org.xbib.elasticsearch.action.ingest;
003    
004    import org.elasticsearch.ElasticsearchIllegalStateException;
005    import org.elasticsearch.action.ActionListener;
006    import org.elasticsearch.action.ActionRequest;
007    import org.elasticsearch.action.WriteConsistencyLevel;
008    import org.elasticsearch.action.delete.DeleteRequest;
009    import org.elasticsearch.action.index.IndexRequest;
010    import org.elasticsearch.action.support.replication.ReplicationType;
011    import org.elasticsearch.client.Client;
012    import org.elasticsearch.common.Nullable;
013    import org.elasticsearch.common.bytes.BytesReference;
014    import org.elasticsearch.common.unit.ByteSizeUnit;
015    import org.elasticsearch.common.unit.ByteSizeValue;
016    import org.elasticsearch.common.unit.TimeValue;
017    import org.xbib.elasticsearch.support.metrics.CounterMetric;
018    
019    import java.util.concurrent.Semaphore;
020    import java.util.concurrent.TimeUnit;
021    import java.util.concurrent.atomic.AtomicLong;
022    
023    public class IngestProcessor {
024    
025        private final Client client;
026    
027        private final int concurrency;
028    
029        private final int actions;
030    
031        private final ByteSizeValue maxVolume;
032    
033        private final TimeValue waitForResponses;
034    
035        private final Semaphore semaphore;
036    
037        private final AtomicLong bulkId;
038    
039        private final IngestRequest ingestRequest;
040    
041        private Listener listener;
042    
043        private volatile boolean closed = false;
044    
045        public IngestProcessor(Client client, Integer concurrency, Integer actions,
046                               ByteSizeValue maxVolume, TimeValue waitForResponses) {
047            this.client = client;
048            this.concurrency = concurrency != null ?
049                    Math.min(Math.abs(concurrency), 256) :
050                    Runtime.getRuntime().availableProcessors() * 4;
051            this.actions = actions != null ? Math.min(actions, 32768) : 1000;
052            this.maxVolume = maxVolume != null ?
053                    new ByteSizeValue(Math.max(maxVolume.bytes(), 1024), ByteSizeUnit.BYTES) :
054                    new ByteSizeValue(10, ByteSizeUnit.MB);
055            this.waitForResponses = waitForResponses != null ?
056                    new TimeValue(Math.max(waitForResponses.millis(), 1000), TimeUnit.MILLISECONDS) :
057                    new TimeValue(60, TimeUnit.SECONDS);
058            this.semaphore = new Semaphore(this.concurrency);
059            this.bulkId = new AtomicLong(0L);
060            this.ingestRequest = new IngestRequest();
061        }
062    
063        public IngestProcessor listener(Listener listener) {
064            this.listener = listener;
065            return this;
066        }
067    
068        public IngestProcessor listenerThreaded(boolean threaded) {
069            ingestRequest.listenerThreaded(threaded);
070            return this;
071        }
072    
073        public IngestProcessor replicationType(ReplicationType type) {
074            ingestRequest.replicationType(type);
075            return this;
076        }
077    
078        public IngestProcessor consistencyLevel(WriteConsistencyLevel level) {
079            ingestRequest.consistencyLevel(level);
080            return this;
081        }
082    
083        /**
084         * Adds an {@link org.elasticsearch.action.index.IndexRequest} to the list
085         * of actions to execute. Follows the same behavior of
086         * {@link org.elasticsearch.action.index.IndexRequest} (for example, if no
087         * id is provided, one will be generated, or usage of the create flag).
088         */
089        public IngestProcessor add(IndexRequest request) {
090            ingestRequest.add((ActionRequest) request);
091            flushIfNeeded(listener);
092            return this;
093        }
094    
095        /**
096         * Adds an {@link org.elasticsearch.action.delete.DeleteRequest} to the list
097         * of actions to execute.
098         */
099        public IngestProcessor add(DeleteRequest request) {
100            ingestRequest.add((ActionRequest) request);
101            flushIfNeeded(listener);
102            return this;
103        }
104    
105        /**
106         * For REST API
107         * @param data the REST body data
108         * @param contentUnsafe if content is unsafe
109         * @param defaultIndex default index
110         * @param defaultType default type
111         * @param listener the listener
112         * @return this processor
113         * @throws Exception
114         */
115        public IngestProcessor add(BytesReference data, boolean contentUnsafe,
116                                   @Nullable String defaultIndex, @Nullable String defaultType,
117                                   Listener listener) throws Exception {
118            ingestRequest.add(data, contentUnsafe, defaultIndex, defaultType);
119            flushIfNeeded(listener);
120            return this;
121        }
122    
123        /**
124         * Closes the processor. If flushing by time is enabled, then it is shut down.
125         * Any remaining bulk actions are flushed, and for the bulk responses is being waited.
126         */
127        public void close() throws InterruptedException {
128            if (closed) {
129                throw new ElasticsearchIllegalStateException("processor already closed");
130            }
131            closed = true;
132            flush();
133            waitForResponses(waitForResponses);
134        }
135    
136        /**
137         * Flush this bulk processor, write all requests
138         */
139        public synchronized void flush() {
140            if (ingestRequest.numberOfActions() > 0) {
141                process(ingestRequest.takeAll(), listener);
142            }
143        }
144    
145        /**
146         * Wait for responses of outstanding requests.
147         *
148         * @return true if all requests answered within the waiting time, false if not
149         * @throws InterruptedException
150         */
151        public boolean waitForResponses(TimeValue maxWait) throws InterruptedException {
152            semaphore.tryAcquire(concurrency, maxWait.getMillis(), TimeUnit.MILLISECONDS);
153            return semaphore.availablePermits() == concurrency;
154        }
155    
156        /**
157         * A listener for the execution.
158         */
159        public static interface Listener {
160    
161            /**
162             * Callback before the bulk request is executed.
163             */
164            void beforeBulk(long bulkId, int concurrency, IngestRequest request);
165    
166            /**
167             * Callback after a successful execution of a bulk request.
168             */
169            void afterBulk(long bulkId, int concurrency, IngestResponse response);
170    
171            /**
172             * Callback after a failed execution of a bulk request.
173             */
174            void afterBulk(long bulkId, int concurrency, Throwable failure);
175        }
176    
177        /**
178         * Critical phase, check if flushing condition is met and
179         * push the part of the bulk requests that is required to push
180         */
181        private synchronized void flushIfNeeded(Listener listener) {
182            if (closed) {
183                throw new ElasticsearchIllegalStateException("processor already closed");
184            }
185            if (actions > 0) {
186                while (ingestRequest.numberOfActions() >= actions) {
187                   process(ingestRequest.take(actions), listener);
188                }
189            } else {
190                while (ingestRequest.numberOfActions() > 0
191                        && maxVolume.bytesAsInt() > 0
192                        && ingestRequest.estimatedSizeInBytes() > maxVolume.bytesAsInt()) {
193                    process(ingestRequest.takeAll(), listener);
194                }
195            }
196        }
197    
198        /**
199         * Process an ingest request and send responses via the listener.
200         *
201         * @param request the ingest request
202         */
203        private void process(final IngestRequest request, final Listener listener) {
204            if (listener == null) {
205                return;
206            }
207            final long id = bulkId.incrementAndGet();
208            boolean done = false;
209            try {
210                semaphore.acquire();
211                listener.beforeBulk(id, concurrency - semaphore.availablePermits(), request);
212                client.execute(IngestAction.INSTANCE, request, new ActionListener<IngestResponse>() {
213                    @Override
214                    public void onResponse(IngestResponse response) {
215                        try {
216                            listener.afterBulk(id, concurrency - semaphore.availablePermits(), response);
217                        } finally {
218                            semaphore.release();
219                        }
220                    }
221    
222                    @Override
223                    public void onFailure(Throwable e) {
224                        try {
225                            listener.afterBulk(id, concurrency - semaphore.availablePermits(), e);
226                        } finally {
227                            semaphore.release();
228                        }
229                    }
230                });
231                done = true;
232            } catch (InterruptedException e) {
233                // semaphore not acquired
234                Thread.currentThread().interrupt();
235                listener.afterBulk(id, concurrency - semaphore.availablePermits(), e);
236            } finally {
237                if (!done) {
238                    semaphore.release();
239                }
240            }
241        }
