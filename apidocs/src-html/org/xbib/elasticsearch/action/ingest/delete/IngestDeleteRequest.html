001    
002    package org.xbib.elasticsearch.action.ingest.delete;
003    
004    import org.elasticsearch.action.ActionRequest;
005    import org.elasticsearch.action.ActionRequestValidationException;
006    import org.elasticsearch.action.WriteConsistencyLevel;
007    import org.elasticsearch.action.delete.DeleteRequest;
008    import org.elasticsearch.action.support.replication.ReplicationType;
009    import org.elasticsearch.common.io.stream.StreamInput;
010    import org.elasticsearch.common.io.stream.StreamOutput;
011    import org.elasticsearch.common.unit.TimeValue;
012    import org.xbib.elasticsearch.action.ingest.index.IngestIndexShardRequest;
013    
014    import java.io.IOException;
015    import java.util.Queue;
016    import java.util.concurrent.atomic.AtomicLong;
017    
018    import static org.elasticsearch.action.ValidateActions.addValidationError;
019    import static org.elasticsearch.common.collect.Queues.newConcurrentLinkedQueue;
020    
021    public class IngestDeleteRequest extends ActionRequest {
022    
023        private static final int REQUEST_OVERHEAD = 50;
024    
025        private final Queue<DeleteRequest> requests = newQueue();
026    
027        private final AtomicLong sizeInBytes = new AtomicLong();
028    
029        private ReplicationType replicationType = ReplicationType.DEFAULT;
030    
031        private WriteConsistencyLevel consistencyLevel = WriteConsistencyLevel.DEFAULT;
032    
033        private TimeValue timeout = IngestIndexShardRequest.DEFAULT_TIMEOUT;
034    
035        private String defaultIndex;
036    
037        private String defaultType;
038    
039        protected Queue<DeleteRequest> newQueue() {
040            return newConcurrentLinkedQueue();
041        }
042    
043        public IngestDeleteRequest setIndex(String index) {
044            this.defaultIndex = index;
045            return this;
046        }
047    
048        public String getIndex() {
049            return defaultIndex;
050        }
051    
052        public IngestDeleteRequest setType(String type) {
053            this.defaultType = type;
054            return this;
055        }
056    
057        public String getType() {
058            return defaultType;
059        }
060    
061        public Queue<DeleteRequest> requests() {
062            return requests;
063        }
064    
065        /**
066         * Adds a list of delete requests to be executed.
067         */
068        public IngestDeleteRequest add(DeleteRequest... requests) {
069            for (DeleteRequest request : requests) {
070                add(request);
071            }
072            return this;
073        }
074    
075        /**
076         * Adds a list of requests to be executed. Either index or delete requests.
077         */
078        public IngestDeleteRequest add(Iterable<DeleteRequest> requests) {
079            for (DeleteRequest request : requests) {
080                add(request);
081            }
082            return this;
083        }
084    
085        /**
086         * Adds an {@link org.elasticsearch.action.index.IndexRequest} to the list of actions to execute. Follows
087         * the same behavior of {@link org.elasticsearch.action.index.IndexRequest} (for example, if no id is
088         * provided, one will be generated, or usage of the create flag).
089         */
090        public IngestDeleteRequest add(DeleteRequest request) {
091            request.beforeLocalFork();
092            return internalAdd(request);
093        }
094    
095        IngestDeleteRequest internalAdd(DeleteRequest request) {
096            requests.offer(request);
097            sizeInBytes.addAndGet(REQUEST_OVERHEAD);
098            return this;
099        }
100    
101        /**
102         * The number of actions in the bulk request.
103         */
104        public int numberOfActions() {
105            // for ConcurrentLinkedQueue, this call is not O(n), and may not be the size of the current list
106            return requests.size();
107        }
108    
109        /**
110         * The estimated size in bytes of the bulk request.
111         */
112        public long estimatedSizeInBytes() {
113            return sizeInBytes.longValue();
114        }
115    
116        /**
117         * Sets the consistency level of write. Defaults to
118         * {@link org.elasticsearch.action.WriteConsistencyLevel#DEFAULT}
119         */
120        public IngestDeleteRequest consistencyLevel(WriteConsistencyLevel consistencyLevel) {
121            this.consistencyLevel = consistencyLevel;
122            return this;
123        }
124    
125        public WriteConsistencyLevel consistencyLevel() {
126            return this.consistencyLevel;
127        }
128    
129        /**
130         * Set the replication type for this operation.
131         */
132        public IngestDeleteRequest replicationType(ReplicationType replicationType) {
133            this.replicationType = replicationType;
134            return this;
135        }
136    
137        public ReplicationType replicationType() {
138            return this.replicationType;
139        }
140    
141        /**
142         * Set the timeout for this operation.
143         */
144        public IngestDeleteRequest timeout(TimeValue timeout) {
145            this.timeout = timeout;
146            return this;
147        }
148    
149        public TimeValue timeout() {
150            return this.timeout;
151        }
152    
153        /**
154         * Take all requests out of this bulk request.
155         *
156         * @return another bulk request
157         */
158        public IngestDeleteRequest takeAll() {
159            IngestDeleteRequest request = new IngestDeleteRequest();
160            while (!requests.isEmpty()) {
161                DeleteRequest deleteRequest = requests.poll();
162                if (deleteRequest != null) {
163                    request.add(deleteRequest);
164                }
165            }
166            return request;
167        }
168    
169        /**
170         * Take a number of requests out of this bulk request and put them
171         * into an array list.
172         *
173         * @param numRequests number of requests
174         * @return another bulk request
175         */
176        public IngestDeleteRequest take(int numRequests) {
177            IngestDeleteRequest request = new IngestDeleteRequest();
178            for (int i = 0; i < numRequests; i++) {
179                DeleteRequest deleteRequest = requests.poll();
180                request.add(deleteRequest);
181                sizeInBytes.addAndGet(-REQUEST_OVERHEAD);
182            }
183            return request;
184        }
185    
186        @Override
187        public ActionRequestValidationException validate() {
188            ActionRequestValidationException validationException = null;
189            if (requests.isEmpty()) {
190                validationException = addValidationError("no requests added", validationException);
191            }
192            return validationException;
193        }
194    
195        @Override
196        public void readFrom(StreamInput in) throws IOException {
197            replicationType = ReplicationType.fromId(in.readByte());
198            consistencyLevel = WriteConsistencyLevel.fromId(in.readByte());
199            timeout = TimeValue.readTimeValue(in);
200            int size = in.readVInt();
201            for (int i = 0; i < size; i++) {
202                DeleteRequest request = new DeleteRequest();
203                request.readFrom(in);
204                requests.add(request);
205            }
206        }
207    
208        @Override
209        public void writeTo(StreamOutput out) throws IOException {
210            out.writeByte(replicationType.id());
211            out.writeByte(consistencyLevel.id());
212            timeout.writeTo(out);
213            out.writeVInt(requests.size());
214            for (ActionRequest request : requests) {
215                request.writeTo(out);
216            }
217        }
218    }
