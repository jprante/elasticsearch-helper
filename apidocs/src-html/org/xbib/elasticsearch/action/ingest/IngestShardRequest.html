002package org.xbib.elasticsearch.action.ingest;
003
004import org.elasticsearch.action.support.replication.ShardReplicationOperationRequest;
005import org.elasticsearch.common.io.stream.StreamInput;
006import org.elasticsearch.common.io.stream.StreamOutput;
007
008import java.io.IOException;
009import java.util.List;
010
011import static org.elasticsearch.common.collect.Lists.newArrayList;
012
013public class IngestShardRequest extends ShardReplicationOperationRequest {
014
015    private int shardId;
016
017    private List<IngestItemRequest> items;
018
019    public IngestShardRequest() {
020    }
021
022    public IngestShardRequest(String index, int shardId, List<IngestItemRequest> items) {
023        this.index = index;
024        this.shardId = shardId;
025        this.items = items;
026    }
027
028    public int shardId() {
029        return shardId;
030    }
031
032    public List<IngestItemRequest> items() {
033        return items;
034    }
035
036    /**
037     * Before we fork on a local thread, make sure we copy over the bytes if they are unsafe
038     */
039    @Override
040    public void beforeLocalFork() {
041        for (IngestItemRequest item : items) {
042            ((ShardReplicationOperationRequest) item.request()).beforeLocalFork();
043        }
044    }
045
046    @Override
047    public void writeTo(StreamOutput out) throws IOException {
048        super.writeTo(out);
049        out.writeVInt(shardId);
050        out.writeVInt(items.size());
051        for (IngestItemRequest item : items) {
052            if (item != null) {
053                out.writeBoolean(true);
054                item.writeTo(out);
055            } else {
056                out.writeBoolean(false);
057            }
058        }
059    }
060
061    @Override
062    public void readFrom(StreamInput in) throws IOException {
063        super.readFrom(in);
064        shardId = in.readVInt();
065        int size = in.readVInt();
066        items = newArrayList();
067        for (int i = 0; i < size; i++) {
068            if (in.readBoolean()) {
069                items.add(IngestItemRequest.readBulkItem(in));
070            }
071        }
072    }
073}
