001    
002    package org.xbib.elasticsearch.action.ingest;
003    
004    import org.elasticsearch.action.ActionResponse;
005    import org.elasticsearch.common.io.stream.StreamInput;
006    import org.elasticsearch.common.io.stream.StreamOutput;
007    import org.elasticsearch.common.unit.TimeValue;
008    
009    import java.io.IOException;
010    import java.util.List;
011    
012    import static org.elasticsearch.common.collect.Lists.newLinkedList;
013    
014    public class IngestResponse extends ActionResponse {
015    
016        private List<IngestItemFailure> failure;
017    
018        private int successSize;
019    
020        private long tookInMillis;
021    
022        public IngestResponse() {
023            this.failure = newLinkedList();
024        }
025    
026        public IngestResponse(int successSize, List<IngestItemFailure> failure, long tookInMillis) {
027            this.successSize = successSize;
028            this.failure = failure;
029            this.tookInMillis = tookInMillis;
030        }
031    
032        public int successSize() {
033            return successSize;
034        }
035    
036        public int failureSize() {
037            return failure.size();
038        }
039    
040        public List<IngestItemFailure> failure() {
041            return failure;
042        }
043    
044        /**
045         * How long the bulk execution took.
046         */
047        public TimeValue took() {
048            return new TimeValue(tookInMillis);
049        }
050    
051        /**
052         * How long the bulk execution took.
053         */
054        public TimeValue getTook() {
055            return took();
056        }
057    
058        /**
059         * How long the bulk execution took in milliseconds.
060         */
061        public long tookInMillis() {
062            return tookInMillis;
063        }
064    
065        /**
066         * How long the bulk execution took in milliseconds.
067         */
068        public long getTookInMillis() {
069            return tookInMillis();
070        }
071    
072        /**
073         * Has anything failed with the execution.
074         */
075        public boolean hasFailures() {
076            return !failure.isEmpty();
077        }
078    
079        public String buildFailureMessage() {
080            StringBuilder sb = new StringBuilder();
081            sb.append("failure in bulk execution:");
082            for (IngestItemFailure f : failure) {
083                sb.append("\n[").append(f.pos()).append("], message [").append(f.message()).append("]");
084            }
085            return sb.toString();
086        }
087    
088        @Override
089        public void readFrom(StreamInput in) throws IOException {
090            successSize = in.readVInt();
091            failure = newLinkedList();
092            int size = in.readVInt();
093            for (int i = 0; i < size; i++) {
094                failure.add(new IngestItemFailure(in.readVInt(), in.readString()));
095            }
096            tookInMillis = in.readVLong();
097        }
098    
099        @Override
100        public void writeTo(StreamOutput out) throws IOException {
101            out.writeVInt(successSize);
102            out.writeVInt(failure.size());
103            for (IngestItemFailure f : failure) {
104                out.writeVInt(f.pos());
105                out.writeString(f.message());
106            }
107            out.writeVLong(tookInMillis);
108        }
109    }
