001    
002    package org.xbib.elasticsearch.support.client;
003    
004    import org.elasticsearch.client.Client;
005    
006    /**
007     * Minimal API for document ingesting. Useful for river implementations.
008     */
009    public interface DocumentIngest {
010    
011        Client client();
012    
013        /**
014         * Set the default index
015         *
016         * @param index the index
017         * @return this TransportClientIndexer
018         */
019        DocumentIngest setIndex(String index);
020    
021        /**
022         * Returns the default index
023         *
024         * @return the index
025         */
026        String getIndex();
027    
028        /**
029         * Set the default type
030         *
031         * @param type the type
032         * @return this TransportClientIndexer
033         */
034        DocumentIngest setType(String type);
035    
036        /**
037         * Returns the default type
038         *
039         * @return the type
040         */
041        String getType();
042    
043        /**
044         * Create document
045         *
046         * @param index
047         * @param type
048         * @param id
049         * @param source
050         * @return this ClientIngest
051         */
052        DocumentIngest createDocument(String index, String type, String id, String source);
053    
054        /**
055         * Index document
056         *
057         * @param index
058         * @param type
059         * @param id
060         * @param source
061         * @return this ClientIngest
062         */
063        DocumentIngest indexDocument(String index, String type, String id, String source);
064    
065        /**
066         * Delete document
067         *
068         * @param index
069         * @param type
070         * @param id
071         * @return this ClientIngest
072         */
073        DocumentIngest deleteDocument(String index, String type, String id);
074    
075        /**
076         * Ensure that all documents arrive.
077         *
078         * @return this ClientIngest
079         */
080        DocumentIngest flush();
081    
082        /**
083         *
084         * Shutdown this client
085         */
086        void shutdown();
087    }
