001    
002    package org.xbib.elasticsearch.support.client;
003    
004    import org.elasticsearch.action.admin.cluster.health.ClusterHealthStatus;
005    import org.elasticsearch.common.unit.TimeValue;
006    
007    import java.io.IOException;
008    import java.io.InputStream;
009    import java.io.Reader;
010    
011    /**
012     * Interface for providing convenient ingest methods.
013     */
014    public interface Ingest extends DocumentIngest {
015    
016        /**
017         * Wait for healthy cluster
018         *
019         * @return this TransportClientIndexer
020         * @throws java.io.IOException
021         */
022        Ingest waitForCluster() throws IOException;
023    
024        Ingest waitForCluster(ClusterHealthStatus status, TimeValue timevalue) throws IOException;
025    
026        int waitForRecovery();
027    
028        /**
029         * Set maximum number of bulk actions
030         *
031         * @param bulkActions maximum number of bulk actions
032         * @return this TransportClientIndexer
033         */
034        Ingest maxBulkActions(int bulkActions);
035    
036        /**
037         * Set maximum concurent bulk requests
038         *
039         * @param maxConcurentBulkRequests maximum number of concurrent bulk requests
040         * @return this Ingest
041         */
042        Ingest maxConcurrentBulkRequests(int maxConcurentBulkRequests);
043    
044        /**
045         * Start bulk mode
046         *
047         * @return this Ingest
048         */
049        Ingest startBulk() throws IOException;
050    
051        /**
052         * Stops bulk mode. Enables refresh.
053         *
054         * @return this Ingest
055         */
056        Ingest stopBulk() throws IOException;
057    
058        /**
059         * Settings for the index
060         * @param key the settings key
061         * @param value the settings value
062         * @return this Ingest
063         */
064        Ingest setting(String key, String value);
065    
066        /**
067         * Settings for the index
068         * @param key the settings key
069         * @param value the settings value
070         * @return this Ingest
071         */
072        Ingest setting(String key, Integer value);
073    
074        /**
075         * Settings for the index
076         * @param key the settings key
077         * @param value the settings value
078         * @return this Ingest
079         */
080        Ingest setting(String key, Boolean value);
081    
082        /**
083         * Settings for the index
084         * @param in the input stream to read the settings from
085         * @return this Ingest
086         */
087        Ingest setting(InputStream in) throws IOException;
088    
089        Ingest addMapping(String type, InputStream in) throws IOException;
090    
091        Ingest addMapping(String type, String mapping);
092    
093        /**
094         * Create a new index
095         *
096         * @return this Ingest
097         */
098        Ingest newIndex();
099    
100        Ingest newMappings();
101    
102        /**
103         * Delete index
104         *
105         * @return this Ingest
106         */
107        Ingest deleteIndex();
108    
109        Ingest deleteMapping(String type);
110    
111        /**
112         * Refresh the index.
113         *
114         * @return
115         */
116        Ingest refresh();
117    
118        /**
119         * Add replica level.
120         *
121         * @param level the replica level
122         * @return number of shards after updating replica level
123         */
124        int updateReplicaLevel(int level) throws IOException;
125    
126        /**
127         * Get the ingested data volume so far.
128         *
129         * @return the volume in bytes
130         */
131        long getVolumeInBytes();
132    
133        Ingest shards(int shards);
134    
135        Ingest replica(int replica);
136    
137    
138    
139    }
