088        /**
089         * Create new client
090         * The URI describes host and port of the node the client should connect to,
091         * with the parameter <tt>es.cluster.name</tt> for the cluster name.
092         *
093         * @param uri the cluster URI
094         * @return this indexer
095         */
096        @Override
097        public IngestDeleteClient newClient(URI uri) {
098            super.newClient(uri, settingsBuilder()
099                    .put("cluster.name", findClusterName(uri))
100                    .put("network.server", false)
101                    .put("node.client", true)
102                    .put("client.transport.sniff", false)
103                    .put("client.transport.ignore_cluster_name", false)
104                    .put("client.transport.ping_timeout", "30s")
105                    .put("client.transport.nodes_sampler_interval", "30s")
106                    .build());
107            resetSettings();
108            IngestDeleteProcessor.Listener listener = new IngestDeleteProcessor.Listener() {
109                @Override
110                public void beforeBulk(long executionId, int concurrency, IngestDeleteRequest request) {
111                    currentIngestNumDocs.inc(request.numberOfActions());
112                    totalIngestSizeInBytes.inc(request.estimatedSizeInBytes());
113                    if (logger.isDebugEnabled()) {
114                        logger.debug("before bulk [{}] of {} items, {} bytes, {} outstanding bulk requests",
115                            executionId, request.numberOfActions(), request.estimatedSizeInBytes(), concurrency);
116                    }
117                }
118    
119                @Override
120                public void afterBulk(long executionId, int concurrency, IngestResponse response) {
121                    totalIngest.inc(response.tookInMillis());
122                    if (logger.isDebugEnabled()) {
123                        logger.debug("after bulk [{}] [{} items succeeded] [{} items failed] [{}ms] {} outstanding bulk requests",
124                                executionId, response.successSize(), response.failure().size(), response.took().millis(), concurrency);
125                    }
126                    if (!response.failure().isEmpty()) {
127                        for (IngestItemFailure f: response.failure()) {
128                            logger.error("after bulk [{}] [{} failure reason: {}", executionId, f.pos(), f.message());
129                        }
130                    } else {
131                        currentIngestNumDocs.dec(response.successSize());
132                        totalIngestNumDocs.inc(response.successSize());
133                    }
134                }
135    
136                @Override
137                public void afterBulk(long executionId, int concurrency, Throwable failure) {
138                    logger.error("after bulk ["+executionId+"] failure", failure);
139                    throwable = failure;
140                    closed = true;
141                }
142            };
143            this.ingestProcessor = new IngestDeleteProcessor(client,
144                    maxConcurrentBulkRequests, maxActionsPerBulkRequest, maxVolume, maxWaitTime)
145                    .listener(listener);
146            this.closed = false;
147            return this;
148        }

281        @Override
282        public IngestDeleteClient delete(String index, String type, String id) {
283            if (closed) {
284                throw new ElasticSearchIllegalStateException("client is closed");
285            }
286            if (logger.isTraceEnabled()) {
287                logger.trace("delete: {}/{}/{}", index, type, id);
288            }
289            DeleteRequest deleteRequest = Requests.deleteRequest(index).type(type).id(id);
290            try {
291                currentIngest.inc();
292                ingestProcessor.add(deleteRequest);
293            } catch (Exception e) {
294                throwable = e;
295                closed = true;
296                logger.error("bulk add of delete failed: " + e.getMessage(), e);
297            } finally {
298                currentIngest.dec();
299            }
300            return this;
301        }

339        public IngestDeleteClient flush() {
340            if (closed) {
341                throw new ElasticSearchIllegalStateException("client is closed");
342            }
343            if (client == null) {
344                logger.warn("no client");
345                return this;
346            }
347            ingestProcessor.flush();
348            return this;
349        }
350    
351        @Override
352        public synchronized void shutdown() {
353            if (closed) {
354                super.shutdown();
355                throw new ElasticSearchIllegalStateException("client is closed");
356            }
357            if (client == null) {
358                logger.warn("no client");
359                return;
360            }
361            try {
362                if (ingestProcessor != null) {
363                    logger.info("closing ingest processor...");
364                    ingestProcessor.close();
365                }
366                logger.info("enabling refresh interval...");
367                enableRefreshInterval();
368                logger.info("shutting down...");
369                super.shutdown();
370                logger.info("shutting down completed");
371            } catch (Exception e) {
372                logger.error(e.getMessage(), e);
373            }
374        }
