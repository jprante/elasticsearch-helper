001    
002    package org.xbib.elasticsearch.support.client;
003    
004    import java.net.URI;
005    
006    import org.elasticsearch.client.Client;
007    import org.elasticsearch.common.settings.ImmutableSettings;
008    import org.elasticsearch.common.settings.Settings;
009    
010    import org.xbib.elasticsearch.action.search.support.BasicRequest;
011    
012    /**
013     * Search client support
014     */
015    public class SearchClient extends AbstractClient implements Search {
016    
017        protected Settings settings;
018    
019        private String index;
020    
021        private String type;
022    
023        public SearchClient setIndex(String index) {
024            this.index = index;
025            return this;
026        }
027    
028        public SearchClient setType(String type) {
029            this.type = type;
030            return this;
031        }
032    
033        @Override
034        public String getIndex() {
035            return index;
036        }
037    
038        @Override
039        public String getType() {
040            return type;
041        }
042    
043        @Override
044        public SearchClient newClient() {
045            super.newClient();
046            return this;
047        }
048    
049        @Override
050        public SearchClient newClient(URI uri) {
051            super.newClient(uri);
052            return this;
053        }
054    
055        public Client client() {
056            return super.client();
057        }
058    
059        /**
060         * Create settings
061         *
062         * @param uri
063         * @param n the client thread pool size
064         * @return the settings
065         */
066        protected Settings initialSettings(URI uri, int n) {
067            return ImmutableSettings.settingsBuilder()
068                    .put("cluster.name", findClusterName(uri))
069                    .put("network.server", false)
070                    .put("node.client", true) // node client
071                    .put("client.transport.sniff", false) // sniff would join us into any cluster ...?
072                    .build();
073        }
074    
075        @Override
076        public BasicRequest newSearchRequest() {
077            return new BasicRequest()
078                    .newSearchRequest(client.prepareSearch().setPreference("_primary_first"));
079        }
080    
081        @Override
082        public BasicRequest newGetRequest() {
083            return new BasicRequest()
084                    .newGetRequest(client.prepareGet());
085        }
086    
087    }
