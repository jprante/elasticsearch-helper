001    package org.xbib.elasticsearch.support.config;
002    
003    import org.elasticsearch.action.admin.indices.mapping.delete.DeleteMappingRequest;
004    import org.elasticsearch.action.admin.indices.mapping.put.PutMappingRequest;
005    import org.elasticsearch.client.Client;
006    import org.elasticsearch.common.io.Streams;
007    import org.elasticsearch.common.settings.ImmutableSettings;
008    import org.elasticsearch.common.settings.Settings;
009    import org.elasticsearch.common.xcontent.XContentBuilder;
010    
011    import java.io.IOException;
012    import java.io.InputStream;
013    import java.io.InputStreamReader;
014    import java.io.StringWriter;
015    import java.util.Map;
016    
017    import static org.elasticsearch.common.collect.Maps.newHashMap;
018    import static org.elasticsearch.common.xcontent.XContentFactory.jsonBuilder;
019    
020    public class ConfigHelper {
021    
022        private ImmutableSettings.Builder settingsBuilder;
023    
024        private Map<String,String> mappings = newHashMap();
025    
026        private boolean dateDetection = false;
027    
028        private boolean timeStampFieldEnabled = false;
029    
030        private boolean kibanaEnabled = false;
031    
032        private String timeStampField = "@timestamp";
033    
034        public ConfigHelper reset() {
035            settingsBuilder = ImmutableSettings.settingsBuilder();
036            return this;
037        }
038    
039        public ConfigHelper setting(String key, String value) {
040            if (settingsBuilder == null) {
041                settingsBuilder = ImmutableSettings.settingsBuilder();
042            }
043            settingsBuilder.put(key, value);
044            return this;
045        }
046    
047        public ConfigHelper setting(String key, Boolean value) {
048            if (settingsBuilder == null) {
049                settingsBuilder = ImmutableSettings.settingsBuilder();
050            }
051            settingsBuilder.put(key, value);
052            return this;
053        }
054    
055        public ConfigHelper setting(String key, Integer value) {
056            if (settingsBuilder == null) {
057                settingsBuilder = ImmutableSettings.settingsBuilder();
058            }
059            settingsBuilder.put(key, value);
060            return this;
061        }
062    
063        public ConfigHelper setting(InputStream in) throws IOException {
064            if (settingsBuilder == null) {
065                settingsBuilder = ImmutableSettings.settingsBuilder();
066            }
067            settingsBuilder = ImmutableSettings.settingsBuilder().loadFromStream(".json", in);
068            return this;
069        }
070    
071        public ImmutableSettings.Builder settingsBuilder() {
072            return settingsBuilder != null ? settingsBuilder : ImmutableSettings.settingsBuilder();
073        }
074    
075        public Settings settings() {
076            if (settingsBuilder == null) {
077                settingsBuilder = ImmutableSettings.settingsBuilder();
078            }
079            return settingsBuilder.build();
080        }
081    
082        public ConfigHelper mapping(String type, InputStream in) throws IOException {
083            if (type == null) {
084                return this;
085            }
086            StringWriter sw = new StringWriter();
087            Streams.copy(new InputStreamReader(in), sw);
088            mappings.put(type, sw.toString());
089            return this;
090        }
091    
092        public ConfigHelper mapping(String type, String mapping) {
093            if (type == null) {
094                return this;
095            }
096            this.mappings.put(type, mapping);
097            return this;
098        }
099    
100        public ConfigHelper putMapping(Client client, String index) {
101            if (!mappings.isEmpty()) {
102                for (Map.Entry<String,String> me : mappings.entrySet()) {
103                    client.admin().indices().putMapping(new PutMappingRequest(index).type(me.getKey()).source(me.getValue())).actionGet();
104                }
105            }
106            return this;
107        }
108    
109        public ConfigHelper deleteMappings(Client client, String index) {
110            if (!mappings.isEmpty()) {
111                for (Map.Entry<String,String> me : mappings.entrySet()) {
112                    client.admin().indices().deleteMapping(new DeleteMappingRequest(index).types(me.getKey())).actionGet();
113                }
114            }
115            return this;
116        }
117    
118        public ConfigHelper deleteMapping(Client client, String index, String type) {
119            client.admin().indices().deleteMapping(new DeleteMappingRequest(index).types(type)).actionGet();
120            return this;
121        }
122    
123        public ConfigHelper dateDetection(boolean dateDetection) {
124            this.dateDetection = dateDetection;
125            return this;
126        }
127    
128        public boolean dateDetection() {
129            return dateDetection;
130        }
131    
132        public ConfigHelper timeStampField(String timeStampField) {
133            this.timeStampField = timeStampField;
134            return this;
135        }
136    
137        public String timeStampField() {
138            return timeStampField;
139        }
140    
141        public ConfigHelper timeStampFieldEnabled(boolean enable) {
142            this.timeStampFieldEnabled = enable;
143            return this;
144        }
145    
146        public ConfigHelper kibanaEnabled(boolean enable) {
147            this.kibanaEnabled = enable;
148            return this;
149        }
150    
151        public String defaultMapping() throws IOException{
152                XContentBuilder b = jsonBuilder()
153                        .startObject()
154                        .startObject("_default_")
155                        .field("date_detection", dateDetection);
156                if (timeStampFieldEnabled) {
157                    b.startObject("_timestamp")
158                            .field("enabled", timeStampFieldEnabled)
159                            .field("path", timeStampField)
160                            .endObject();
161                }
162                if (kibanaEnabled) {
163                    b.startObject("properties")
164                            .startObject("@fields")
165                            .field("type", "object")
166                            .field("dynamic", true)
167                            .field("path", "full")
168                            .endObject()
169                            .startObject("@message")
170                            .field("type", "string")
171                            .field("index", "analyzed")
172                            .endObject()
173                            .startObject("@source")
174                            .field("type", "string")
175                            .field("index", "not_analyzed")
176                            .endObject()
177                            .startObject("@source_host")
178                            .field("type", "string")
179                            .field("index", "not_analyzed")
180                            .endObject()
181                            .startObject("@source_path")
182                            .field("type", "string")
183                            .field("index", "not_analyzed")
184                            .endObject()
185                            .startObject("@tags")
186                            .field("type", "string")
187                            .field("index", "not_analyzed")
188                            .endObject()
189                            .startObject("@timestamp")
190                            .field("type", "date")
191                            .endObject()
192                            .startObject("@type")
193                            .field("type", "string")
194                            .field("index", "not_analyzed")
195                            .endObject()
196                            .endObject();
197                }
198                b.endObject()
199                        .endObject();
200                return b.string();
201        }
202    
203        public Map<String, String> mappings() {
204            return mappings.isEmpty() ? null : mappings;
205        }
206    
207    }
