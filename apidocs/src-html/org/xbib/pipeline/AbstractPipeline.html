001package org.xbib.pipeline;
002
003import org.xbib.metrics.MeterMetric;
004
005import java.util.Iterator;
006import java.util.LinkedHashMap;
007import java.util.Map;
008import java.util.concurrent.TimeUnit;
009
010/**
011 * Basic pipeline for pressing pipeline requests.
012 * This abstract class can be used for creating custom Pipeline classes.
013 *
014 * @param <R> the pipeline request type
015 * @param <E> the pipeline error type
016 */
017public abstract class AbstractPipeline<R extends PipelineRequest, E extends PipelineException>
018        implements Pipeline<MeterMetric, R>, PipelineRequestListener<MeterMetric, R>,
019        PipelineErrorListener<MeterMetric, R, E> {
020
021    /**
022     * A list of request listeners for processing requests
023     */
024    private Map<String, PipelineRequestListener<MeterMetric, R>> requestListeners =
025            new LinkedHashMap<String, PipelineRequestListener<MeterMetric, R>>();
026
027    /**
028     * A list of error listeners for processing errors
029     */
030    private Map<String, PipelineErrorListener<MeterMetric, R, E>> errorListeners =
031            new LinkedHashMap<String, PipelineErrorListener<MeterMetric, R, E>>();
032
033    private MeterMetric metric;
034
035    /**
036     * Add a pipeline request listener to the pipeline. The listener is called each time
037     * this pipeline processes a new request.
038     *
039     * @param name     the listener name
040     * @param listener the listener
041     * @return this pipeline
042     */
043    public Pipeline<MeterMetric, R> add(String name, PipelineRequestListener<MeterMetric, R> listener) {
044        if (name != null) {
045            this.requestListeners.put(name, listener);
046        }
047        return this;
048    }
049
050    /**
051     * Add a pipeline request listener to the pipeline. The listener is called each time
052     * this pipeline processes a new request.
053     *
054     * @param name     the listener name
055     * @param listener the listener
056     * @return this pipeline
057     */
058    public Pipeline<MeterMetric, R> add(String name, PipelineErrorListener<MeterMetric, R, E> listener) {
059        if (name != null) {
060            this.errorListeners.put(name, listener);
061        }
062        return this;
063    }
064
065    /**
066     * Call this thread. Iterate over all request and pass them to request listeners.
067     * At least, this pipeline itself can listen to requests and handle errors.
068     * Only PipelineExceptions are handled for each listener. Other execptions will quit the
069     * pipeline request executions.
070     *
071     * @return a metric about the pipeline request executions.
072     * @throws Exception if pipeline execution was sborted by a non-PipelineException
073     */
074    @Override
075    public MeterMetric call() throws Exception {
076        try {
077            metric = new MeterMetric(5L, TimeUnit.SECONDS);
078            Iterator<R> it = this;
079            while (it.hasNext()) {
080                R r = it.next();
081                // add ourselves if not already done
082                requestListeners.put(null, this);
083                errorListeners.put(null, this);
084                for (PipelineRequestListener<MeterMetric, R> requestListener : requestListeners.values()) {
085                    try {
086                        requestListener.newRequest(this, r);
087                    } catch (PipelineException e) {
088                        for (PipelineErrorListener<MeterMetric, R, E> errorListener : errorListeners.values()) {
089                            errorListener.error(this, r, (E) e);
090                        }
091                    }
092                }
093                metric.mark();
094            }
095            close();
096        } finally {
097            metric.stop();
098        }
099        return getMetric();
100    }
101
102    /**
103     * Removing pipeline requests is not supported.
104     */
105    @Override
106    public void remove() {
107        throw new UnsupportedOperationException("remove not supported");
108    }
109
110    /**
111     * Return the metric.
112     *
113     * @return the metric of this pipeline
114     */
115    public MeterMetric getMetric() {
116        return metric;
117    }
118
119    /**
120     * A new request for the pipeline is processed.
121     *
122     * @param pipeline the pipeline
123     * @param request  the pipeline request
124     */
125    public abstract void newRequest(Pipeline<MeterMetric, R> pipeline, R request);
126
127    /**
128     * A PipelineException occured.
129     *
130     * @param pipeline the pipeline
131     * @param request  the pipeline request
132     * @param error    the pipeline error
133     */
134    public abstract void error(Pipeline<MeterMetric, R> pipeline, R request, E error);
135
136}
