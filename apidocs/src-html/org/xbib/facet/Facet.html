019    package org.xbib.facet;
020    
021    import java.util.ArrayList;
022    import java.util.List;
023    
024    /**
025     *  Facet
026     *  
027     * @author <a href="mailto:joergprante@gmail.com">J&ouml;rg Prante</a>
028     */
029    public class Facet {
030    
031        private String displayLabel;
032    
033        private String description;
034    
035        private String index;
036    
037        private String relation;
038    
039        private List<FacetTerm> terms = new ArrayList();
040    
041        public Facet(String displayLabel, String description, String index, String relation) {
042            this.displayLabel = displayLabel;
043            this.description = description;
044            this.index = index;
045            this.relation = relation;
046            this.terms = new ArrayList();
047        }
048    
049        public void add(FacetTerm term) {
050            terms.add(term);
051        }
052    
053        public String getDisplayLabel() {
054            return displayLabel;
055        }
056    
057        public String getDescription() {
058            return description;
059        }
060    
061        public String getIndex() {
062            return index;
063        }
064    
065        public String getRelation() {
066            return relation;
067        }
068    
069        public List<FacetTerm> getTerms() {
070            return terms;
071        }
072    
073    }
