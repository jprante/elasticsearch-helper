121    protected Node startNode(String id) {
122        return buildNode(id).start();
123    }
124
125    private Node buildNode(String id) {
126        String settingsSource = getClass().getName().replace('.', '/') + ".yml";
127        Settings finalSettings = settingsBuilder()
128                .loadFromClasspath(settingsSource)
129                .put(getNodeSettings())
130                .put("name", id)
131                .build();
132        Node node = nodeBuilder().settings(finalSettings).build();
133        Client client = node.client();
134        nodes.put(id, node);
135        clients.put(id, client);
136        return node;
137    }
138
139    protected void stopNode(String id) {
140        Client client = clients.remove(id);
141        if (client != null) {
142            client.close();
143        }
144        Node node = nodes.remove(id);
145        if (node != null) {
146            node.close();
147        }
148    }
149
150    public Client client(String id) {
151        return clients.get(id);
152    }
153
154    public void closeAllNodes() {
155        for (Client client : clients.values()) {
156            client.close();
157        }
158        clients.clear();
159        for (Node node : nodes.values()) {
160            if (node != null) {
161                node.close();
162            }
163        }
164        nodes.clear();
165    }
