103            logger.info("total bulk requests = {}", ingest.getMetric().getTotalIngest().count());
104            assertEquals(1, ingest.getMetric().getTotalIngest().count());
105            if (ingest.hasThrowable()) {
106                logger.error("error", ingest.getThrowable());
107            }
108            assertFalse(ingest.hasThrowable());

110        }

115        Settings settings = Settings.settingsBuilder()
116                .put("index.number_of_shards", 2)
117                .put("index.number_of_replicas", 1)
118                .build();
119        final IngestTransportClient ingest = new IngestTransportClient()
120                .flushIngestInterval(TimeValue.timeValueSeconds(600))
121                .maxActionsPerRequest(MAX_ACTIONS)
122                .newClient(getSettings())
123                .newIndex("test", settings, null)
124                .startBulk("test", -1, 1000);
125        try {
126            for (int i = 0; i < NUM_ACTIONS; i++) {
127                ingest.index("test", "test", null, "{ \"name\" : \"" + randomString(32) + "\"}");
128            }
129            ingest.flushIngest();
130            ingest.waitForResponses(TimeValue.timeValueSeconds(30));
131        } catch (NoNodeAvailableException e) {
132            logger.warn("skipping, no node available");
133        } catch (InterruptedException e) {
134            // ignore
135        } finally {
136            ingest.stopBulk("test");
137            logger.info("total requests = {}", ingest.getMetric().getTotalIngest().count());
138            assertEquals(NUM_ACTIONS / MAX_ACTIONS + 1, ingest.getMetric().getTotalIngest().count());
139            if (ingest.hasThrowable()) {

147    @Test
148    public void testThreadedRandomDocsIngestClient() throws Exception {
149        int maxthreads = Runtime.getRuntime().availableProcessors();
150        int maxactions = MAX_ACTIONS;
151        final int maxloop = NUM_ACTIONS;
152        Settings settings = Settings.settingsBuilder()
153                .put("index.number_of_shards", 2)
154                .put("index.number_of_replicas", 1)
155                .build();
156        final IngestTransportClient ingest = new IngestTransportClient()
157                .flushIngestInterval(TimeValue.timeValueSeconds(600))
158                .maxActionsPerRequest(maxactions)
159                .newClient(getSettings())
160                .newIndex("test", settings, null)
161                .startBulk("test", -1, 1000);
162        try {
163            ThreadPoolExecutor pool = EsExecutors.newFixed(maxthreads, 30,
164                    EsExecutors.daemonThreadFactory("ingest-test"));
165            final CountDownLatch latch = new CountDownLatch(maxthreads);
166            for (int i = 0; i < maxthreads; i++) {
167                pool.execute(new Runnable() {
168                    public void run() {
169                        for (int i = 0; i < maxloop; i++) {
170                            ingest.index("test", "test", null, "{ \"name\" : \"" + randomString(32) + "\"}");
171                        }
172                        latch.countDown();
173                    }
174                });
175            }
176            logger.info("waiting for max 60 seconds...");
177            latch.await(60, TimeUnit.SECONDS);
178            logger.info("client flush ...");
179            ingest.flushIngest();
180            ingest.waitForResponses(TimeValue.timeValueSeconds(30));
181            logger.info("thread pool to be shut down ...");
182            pool.shutdown();
183            logger.info("thread pool shut down");
184        } catch (NoNodeAvailableException e) {
185            logger.warn("skipping, no node available");
186        } finally {
187            ingest.stopBulk("test");
188            logger.info("total requests = {}", ingest.getMetric().getTotalIngest().count());
189            assertEquals(maxthreads * maxloop / maxactions + 1, ingest.getMetric().getTotalIngest().count());
190            if (ingest.hasThrowable()) {
191                logger.error("error", ingest.getThrowable());
192            }
193            assertFalse(ingest.hasThrowable());
194            ingest.refreshIndex("test");
195            assertEquals(maxthreads * maxloop,
196                    ingest.client().prepareCount("test").setQuery(QueryBuilders.matchAllQuery()).execute().actionGet().getCount()
197            );
198            ingest.shutdown();
199        }
200    }

202    @Test
203    public void testClusterConnect() throws IOException {
204        startNode("2");
205        Settings.Builder settingsBuilder = Settings.builder()
206                .put("cluster.name", getClusterName())
207                .put("path.home", System.getProperty("path.home"))
208                .put("autodiscover", true);
209        int i = 0;
210        NodesInfoRequest nodesInfoRequest = new NodesInfoRequest().transport(true);
211        NodesInfoResponse response = client("1").admin().cluster().nodesInfo(nodesInfoRequest).actionGet();
212        for (NodeInfo nodeInfo : response) {
213            TransportAddress ta = nodeInfo.getTransport().getAddress().publishAddress();
214            if (ta instanceof InetSocketTransportAddress) {
215                InetSocketTransportAddress address = (InetSocketTransportAddress) ta;
216                settingsBuilder.put("host." + i++, address.address().getHostName() + ":" + address.address().getPort());
217            }
218        }
219        final IngestTransportClient ingest = new IngestTransportClient()
220                .newClient(settingsBuilder.build())
221                .newIndex("test");
222        ingest.shutdown();
223        if (ingest.hasThrowable()) {
224            logger.error("error", ingest.getThrowable());
225        }
226        assertFalse(ingest.hasThrowable());
227    }
