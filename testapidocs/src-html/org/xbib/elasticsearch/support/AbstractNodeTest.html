002package org.xbib.elasticsearch.support;
003
004import java.util.Map;
005
006import org.elasticsearch.action.admin.cluster.node.info.NodesInfoRequest;
007import org.elasticsearch.action.admin.cluster.node.info.NodesInfoResponse;
008import org.elasticsearch.action.admin.indices.create.CreateIndexRequest;
009import org.elasticsearch.action.admin.indices.delete.DeleteIndexRequest;
010import org.elasticsearch.client.Client;
011import org.elasticsearch.common.logging.ESLogger;
012import org.elasticsearch.common.logging.ESLoggerFactory;
013import org.elasticsearch.common.network.NetworkUtils;
014import org.elasticsearch.common.settings.ImmutableSettings;
015import org.elasticsearch.common.settings.Settings;
016import org.elasticsearch.common.transport.InetSocketTransportAddress;
017import org.elasticsearch.indices.IndexMissingException;
018import org.elasticsearch.node.Node;
019import org.testng.Assert;
020import org.testng.annotations.AfterMethod;
021import org.testng.annotations.BeforeMethod;
022
023import static org.elasticsearch.common.collect.Maps.newHashMap;
024import static org.elasticsearch.common.settings.ImmutableSettings.Builder.EMPTY_SETTINGS;
025import static org.elasticsearch.common.settings.ImmutableSettings.settingsBuilder;
026import static org.elasticsearch.node.NodeBuilder.nodeBuilder;
027
028public abstract class AbstractNodeTest extends Assert {
029
030    private final static ESLogger logger = ESLoggerFactory.getLogger("test");
031
032    protected final String CLUSTER = "test-cluster-" + NetworkUtils.getLocalAddress().getHostName();
033
034    protected final String INDEX = "test-" + NetworkUtils.getLocalAddress().getHostName().toLowerCase();
035
036    protected int PORT;
037
038    protected Settings defaultSettings = ImmutableSettings
039            .settingsBuilder()
040            .put("cluster.name", CLUSTER)
041                    // default for queue_size for bulk thread pool is only 50 since 0.90.7
042                    // enlarge queue for this unit test because we are on a single machine
043            .put("threadpool.bulk.queue_size", 1000)
044            .build();
045
046    private Map<String, Node> nodes = newHashMap();
047
048    private Map<String, Client> clients = newHashMap();
049
050    private Map<String, InetSocketTransportAddress> addresses = newHashMap();
051
052    @BeforeMethod
053    public void createIndex() throws Exception {
054        startNode("1");
055        // find node address
056        NodesInfoRequest nodesInfoRequest = new NodesInfoRequest().transport(true);
057        NodesInfoResponse response = client("1").admin().cluster().nodesInfo(nodesInfoRequest).actionGet();
058        InetSocketTransportAddress address = (InetSocketTransportAddress)response.iterator().next()
059                        .getTransport().getAddress().publishAddress();
060        PORT = address.address().getPort();
061        addresses.put("1", address);
062        logger.info("creating index {}", INDEX);
063        client("1").admin().indices().create(new CreateIndexRequest(INDEX)).actionGet();
064        logger.info("index {} created", INDEX);
065    }
066
067    @AfterMethod
068    public void deleteIndices() {
069        logger.info("deleting index {}", INDEX);
070        try {
071            client("1").admin().indices().delete(new DeleteIndexRequest().indices(INDEX)).actionGet();
072        } catch (IndexMissingException e) {
073            // ignore
074        }
075        logger.info("index {} deleted", INDEX);
076        closeAllNodes();
077    }
078
079    public Node startNode(String id) {
080        return buildNode(id).start();
081    }
082
083    public Node buildNode(String id) {
084        return buildNode(id, EMPTY_SETTINGS);
085    }
086
087    public Node buildNode(String id, Settings settings) {
088        String settingsSource = getClass().getName().replace('.', '/') + ".yml";
089        Settings finalSettings = settingsBuilder()
090                .loadFromClasspath(settingsSource)
091                .put(defaultSettings)
092                .put(settings)
093                .put("name", id)
094                .build();
095        if (finalSettings.get("gateway.type") == null) {
096            finalSettings = settingsBuilder().put(finalSettings).put("gateway.type", "none").build();
097        }
098        if (finalSettings.get("cluster.routing.schedule") != null) {
099            finalSettings = settingsBuilder().put(finalSettings).put("cluster.routing.schedule", "50ms").build();
100        }
101        Node node = nodeBuilder().settings(finalSettings).build();
102        Client client = node.client();
103        nodes.put(id, node);
104        clients.put(id, client);
105        return node;
106    }
107
108    public void stopNode(String id) {
109        Client client = clients.remove(id);
110        if (client != null) {
111            client.close();
112        }
113        Node node = nodes.remove(id);
114        if (node != null) {
115            node.close();
116        }
117    }
118
119    public Client client(String id) {
120        return clients.get(id);
121    }
122
123    public void closeAllNodes() {
124        for (Client client : clients.values()) {
125            client.close();
126        }
127        clients.clear();
128        for (Node node : nodes.values()) {
129            node.close();
130        }
131        nodes.clear();
132    }
133
134}
